    virtualenv¶

    # 构建
cd df && docker build . -t ylin_python:base-3.8

# 创建虚拟环境
virtualenv env_name

# 激活虚拟环境
source ./bin/activate

# 查看当前库
pip list

# 退出虚拟环境
deactivate

# 删除虚拟环境
rmvirtualenv test

# 冻结环境依赖
pip freeze > requirements.txt

# 安装环境依赖
pip install -r requirements.txt

    image: python:3
    ports:
      - 4000:8000

      - ./src/python3.7/base:/usr/local/lib/python3.7
      - ./src/projects/flask:/project
    working_dir: /project
    command:
      ["tail","-f","/var/log/btmp"]

    pypy¶

    一样的道理:

    FROM pypy:3

WORKDIR /usr/src/app

COPY requirements.txt ./
RUN pip install --no-cache-dir -r requirements.txt

COPY . .

CMD [ "pypy3", "./your-daemon-or-script.py" ]

    docker run -it --rm --name my-running-script -v "$PWD":/usr/src/myapp -w /usr/src/myapp pypy:3 pypy3 your-daemon-or-script.p
