    Git安装¶

    yum remove git &&
yum install -y curl-devel expat-devel gettext-devel openssl-devel zlib-devel &&

yum install -y gcc perl-ExtUtils-MakeMaker &&
cd /usr/src &&
# wget https://forensics.cert.org/cert-forensics-tools-release-el7.rpm &&
# yum install cert-forensics-tools-release-el7.rpm &&
wget https://www.kernel.org/pub/software/scm/git/git-2.7.3.tar.gz  &&
tar xzf git-2.7.3.tar.gz &&
cd git-2.7.3 &&
make prefix=/usr/local/git all &&
make prefix=/usr/local/git install &&
# ./configure --without-iconv &&
# make CFLAGS=-liconv prefix=/usr/local/git all &&
# make CFLAGS=-liconv prefix=/usr/local/git install &&
cd /usr/bin &&
ln -s  /usr/local/git/bin/git git &&
git --version

    常用命令¶

    通用

    # 用户配置
git config --global user.name "林宇靖"
git config --global user.email "lprintf@qq.com"

# 项目初始化
cd existing_folder
git init
git remote add origin http://git.fvcom.org/ylin/my_profile.git
git add .
git commit
git push -u origin master

# 需改远程仓库地址
git remote set-url origin URL

# 查看远程仓库地址
git remote -v

# 其它
git remote rm origin
git remote add origin

    技巧

    # 删除提交记录
git checkout --orphan latest_branch
git add -A
git commit -am "commit message"
git branch -D master
git branch -m master
git push -f origin master

    gitlab部署¶

    gitlab:
    image: 'gitlab/gitlab-ce:latest'
    restart: always
    hostname: 'gitlab.ylin.com'
    environment:
      GITLAB_OMNIBUS_CONFIG: |
        external_url 'http://gitlab.ylin.com:7080'
        gitlab_rails['gitlab_shell_ssh_port'] = 7022
        # Add any other gitlab.rb configuration here, each on its own line
    ports:
      - '7080:7080'
    #   - '7443:443'
      - '7022:22'
    volumes:
      - ./src/config:/etc/gitlab
      - ./src/logs:/var/log/gitlab
      - ./src/data:/var/opt/gitlab

    注意:

    • 部署耗时较长,请耐心等待

    • 部署后需要做域名解析

    • 由于端口不足才禁用https,端口充足请重配置.

    参考https://docs.gitlab.com/omnibus/docker/#install-gitlab-using-docker-compose

    开启gitlab pages¶

    修改gitlab.rb如下

    ##! Define to enable GitLab Pages
pages_external_url "http://ylin.com/"
gitlab_pages['enable'] = true

    操作没这么简单,不建议开启,详见https://docs.gitlab.com/ee/administration/pages/index.html
